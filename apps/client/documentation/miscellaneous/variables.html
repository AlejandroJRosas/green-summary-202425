src/app/guards/auth.guard.ts

authGuard
Type : CanActivateFn
Default value : () => { const authService = inject(AuthService) const router = inject(Router) if (authService.isAuth()) { return true } else { const url = router.createUrlTree(['/login']) const toast = inject(Toast) toast.show('error', 'Error', '¡Ups, parece que no has iniciado sesión aún!') return url } }

src/app/interceptors/authorization.interceptor.ts

authorizationInterceptor
Type : HttpInterceptorFn
Default value : (req, next) => { const token = localStorage.getItem('token') ?? '' const requestWithHeader = req.clone({ headers: req.headers.set('Authorization', `Bearer ${token}`) }) return next(requestWithHeader) }
